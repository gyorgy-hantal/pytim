Testsuite¶

Module: testsuite1¶

class pytim.tests.testsuite1.Testsuite1¶

This is a collection of basic tests to check that code is running – no test on the correctness of the output is performed here.

>>> # TEST:1 basic functionality
>>> import MDAnalysis as mda
>>> import pytim
>>> from pytim import  *
>>> from pytim.datafiles import *
>>> u         = mda.Universe(WATER_GRO)
>>> oxygens   = u.select_atoms("name OW")
>>> interface = pytim.ITIM(u, alpha=2.0, max_layers=4)
>>> print len(interface.layers[0,0])
780
>>> del interface
>>> interface = pytim.ITIM(u, alpha=2.0, max_layers=4, multiproc=False)
>>> print len(interface.layers[0,0])
780
>>> del interface

>>> u=None
